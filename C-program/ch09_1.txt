#include <stdio.h>
#include <stdlib.h>
int main(void)
{
	int n=0, m=0, i=0, j=0;
	scanf("%d %d", &n, &m);
	int score[n][m];
	for(i=0; i<n; i++)
	{
		for(j=0; j<m; j++)
		{
			scanf("%d", &score[i][j]);
		}
	}
	for(i=0; i<n; i++)
	{
		int min = score[i][0];
		int max = score[i][0];
		int min_id = 0;
		int max_id = 0;

		for(j=0; j<m; j++)
		{
			if(score[i][j] < min)
			{
				min = score[i][j];
				min_id = j;
			}

			if(score[i][j] > max)
			{
				max = score[i][j];
				max_id = j;
			}
		}

		printf("%d %d\n", min_id+1, max_id+1);
	}
	system("pause");
	return 0;
}
